function generateCacheName(kind, id) {
  return kind+'_'+id;

    let _this = this;

        return deferred.resolve({models:[], nextQuery: null});

      let models = [];
      resp.entities.forEach(function(entity){
        models.push(new _this.prototype.constructor(entity));
      });
 
      return deferred.resolve({models:models, nextQuery: resp.nextQuery});

      deferred.reject(new Error(err));
    }));

  }
